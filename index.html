const canvas = document.getElementById('simCanvas'); const ctx = canvas.getContext('2d'); const dpr = window.devicePixelRatio || 1; // Imposto dimensioni canvas con scaling per qualità HiDPI function resizeCanvas() { const rect = canvas.getBoundingClientRect(); canvas.width = rect.width * dpr; canvas.height = rect.height * dpr; ctx.setTransform(1, 0, 0, 1, 0, 0); // reset transform ctx.scale(dpr, dpr); } resizeCanvas(); window.addEventListener('resize', resizeCanvas); // Particelle demo class Particle { constructor(x, y) { this.x = x; this.y = y; this.radius = 8; this.color = 'rgba(30,144,255,0.7)'; this.vx = (Math.random() - 0.5) * 1.5; this.vy = (Math.random() - 0.5) * 1.5; } draw(ctx) { ctx.beginPath(); ctx.fillStyle = this.color; ctx.shadowColor = 'rgba(0,0,0,0.3)'; ctx.shadowBlur = 5; ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2); ctx.fill(); ctx.shadowBlur = 0; } update(width, height) { this.x += this.vx; this.y += this.vy; // Rimbalzo ai bordi if(this.x < this.radius || this.x > width - this.radius) this.vx *= -1; if(this.y < this.radius || this.y > height - this.radius) this.vy *= -1; } } // Setup particelle const particles = []; const PARTICLE_COUNT = 30; function initParticles() { particles.length = 0; const width = canvas.clientWidth; const height = canvas.clientHeight; for(let i=0; i < PARTICLE_COUNT; i++) { const x = Math.random() * (width - 20) + 10; const y = Math.random() * (height - 20) + 10; particles.push(new Particle(x, y)); } } initParticles(); // Drag & drop variabili let draggingParticle = null; let offsetX = 0; let offsetY = 0; canvas.addEventListener('mousedown', (e) => { const rect = canvas.getBoundingClientRect(); const mouseX = e.clientX - rect.left; const mouseY = e.clientY - rect.top; // Controllo se clicco su una particella for(const p of particles) { const dx = mouseX - p.x; const dy = mouseY - p.y; if(Math.sqrt(dx*dx + dy*dy) < p.radius) { draggingParticle = p; offsetX = dx; offsetY = dy; break; } } }); canvas.addEventListener('mousemove', (e) => { if(draggingParticle) { const rect = canvas.getBoundingClientRect(); draggingParticle.x = e.clientX - rect.left - offsetX; draggingParticle.y = e.clientY - rect.top - offsetY; } }); canvas.addEventListener('mouseup', () => { draggingParticle = null; }); canvas.addEventListener('mouseleave', () => { draggingParticle = null; }); // Animazione function animate() { const width = canvas.clientWidth; const height = canvas.clientHeight; ctx.clearRect(0, 0, width, height); for(const p of particles) { if(p !== draggingParticle) { p.update(width, height); } p.draw(ctx); } // Caption update (opzionale) document.getElementById('caption').textContent = "Demo: Fluidodinamica — usa il mouse per spostare le particelle."; requestAnimationFrame(animate); } animate(); // Reset button document.getElementById('btn-reset').addEventListener('click', () => { initParticles(); });
